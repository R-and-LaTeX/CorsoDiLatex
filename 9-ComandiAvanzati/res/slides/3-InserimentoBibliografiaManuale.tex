Nel caso di \texttt{book} o \texttt{report}
\begin{lstlisting}
\cleardoublepage
%\phantomsection
\addcontentsline{toc}{chapter}{\bibname}
\end{lstlisting}

\vfill

Nel caso di \texttt{article}
\begin{lstlisting}
\clearpage
%\phantomsection
\addcontentsline{toc}{section}{\refname}
\end{lstlisting}

\end{frame}
